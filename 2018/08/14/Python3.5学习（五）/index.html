1.5 import优化

1.使用import module来调用模块时，每次module.func()都会寻找一次模块。
而使用from module import func时候，调用func不同寻找模块直接调用，需要注意的是调用的函数名不要与程序中的有重复的。

2.在做程序的时候我们常常采用软件目录的规范来写程序，但是就会出现import的时候添加在搜索路径列表中的路径过多，可能存在相同的模块名在不同的路径中出现，在调用的时候系统会调用先添加进去的模块，就会出现错误。

建议：个人感觉比较好的解决办法是通过将软件整体的目录所在的路径添加到列表中,之后都采用包调用的方式来调用不同目录中的模块，这样我们只需要在程序的主入口添加路径，其他地方都通过from package import moudle的方式调用即可。这样有效避免的搜索路径的困扰

2.3.1 实例：验证码生成器

#随机生成包含数字和字母的验证码
import random

n = int(input("请输入要定义几位的验证码"))
while True:
    #创建空验证码
    checkcode = ""
    #判断是否存在字符
    isstr = False
    for i in range(n):
        current = random.randrange(0,n)
        #当current值为其所在位置编号时，则输出字母
        if current == i:
            tmp = chr(random.randint(65,90))
            isstr = True
        else:
            tmp = str(random.randint(0,9))
        checkcode += tmp
    #判断：如果验证码中存在字母打印并退出
    if isstr:
        print(checkcode)
        break

2.10 shutil

#高级的文件、文件夹、压缩包 处理模块
import shutil
#copyfileobj(文件句柄1,文件句柄2)将文件1拷贝到文件2
f1 = open("file",encoding='utf-8')
f2 = open("file_copy",'w',encoding='utf-8')
shutil.copyfileobj(f1,f2)
f1.close()
f2.close()
#copyfile(文件名1,文件名2)直接输入文件名就可以拷贝
shutil.copyfile("file","file_copy2")
#copymode(文件1，文件2)仅copy权限，不copy用户、属组内容
#copystat():拷贝状态信息不copy内容
shutil.copystat("file","file_copy2")
#copy:拷贝文件和权限
#copy2:拷贝文件和状态信息
#copytree()递归的copy,完整copy目录
shutil.copytree("module","shutil_module")
#rmtree()递归的删除目录,全部删除目录下的所有东西

#匹配字符串
import re
#match 匹配,\d+后面取数字，直到取到字符
res = re.match("^张三\d+",'张三213来啊')
print(res.group())
#'.'默认匹配除\n之外的任意字符，若指定flag DOTALL则匹配任意字符，包括换行
res2 = re.match("^.+\d",'张三213来啊\n666')
print(res2.group())
#'^'匹配字符开头,若指定flags MULTILINE,也可匹配上(r"^a","\nabc\nee")
#'.'
res3 = re.match("^","张三0342来不来")
print(res3)
#match:从头匹配
#search使用最多
#a$表示匹配到以a为结尾，$判断是不是以某一个字符结尾
res4 = re.search("3.+$","张三0342来不来")
print(res4)
#a.+b表示从a匹配到b
#"R[a-z]+a"表示从R取字母一直到第一个a
res5 = re.search("R[a-z]+i","CzRsdjfi45a")
print(res5)
#"R[a-zA-Z]+a"可以取到R-a之间的大小写字母
#ab*:匹配*前的数字一次或多次,findall返回全部匹配值
res6 =re.findall("ab*","dfiabdueabbab")
print(res6)
#ab?匹配?前1个字符一次或0次，表示b不确定有，先匹配到a或者ab都可以返回
res7 = re.search("ab?","aaaababbbaa")
print(res7)
#{n}表示想匹配几个，[0-9]{3}表示要匹配连续的三个0-9数字
res8 =  re.search("[a-z]{4}","324sdnijf354")
print(res8)
#{a,b}表示匹配连续的字母，最多b次，最少a次
res8 =  re.findall("[a-z]{1,4}","sdf324sdnijf354")
print(res8)
#|表示匹配|左面或者|右面的字符
res9 = re.search("abc|ACB","abcda|BACBD")
print(res9.group())
#(...)分组匹配
res10 = re.search("(abc){2}(\|\|=){2}","abcabc||=||=dddf")
print(res10.group())
