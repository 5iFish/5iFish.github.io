三、实例一：购物车

需求：

  1. 启动程序后，让用户输入工资，然后打印商品列表
  2. 允许用户根据商品编号购买商品
  3. 用户选择商品后，检测余额是否够，够就直接扣款，不够就提醒
  4. 可随时退出，退出时，打印已购买的商品和余额

自己写的代码如下：

# Author:ZT

#创建商品列表
goods = [[1,"红米Note5",1099],
         [2,"华为荣耀畅玩7c",899],
         [3,"洁柔抽纸",43],
         [4,"小熊面包机",199],
         [5,"小米体脂称",179]]
shoppingcart = ["购物清单为："]
#输入薪水
balance = int(input("薪水："))
#进入循环打印商品列表
while balance>0:
    for i in goods:
        print(i)
    numbers =input("想买哪个?")
    if (numbers!='q')and(1<=int(numbers)<=5):
        if balance>=goods[int(numbers)-1][2]:
            shoppingcart.append(goods[int(numbers)-1])
            balance=balance-goods[int(numbers)-1][2]
            print("添加%s到购物车成功！余额为：%d"%(goods[int(numbers)-1][1],balance))
        else:
            print("余额不足！！！")
    else:
        for i in shoppingcart:
            print(i)
        print("余额为：%d"%balance)
        break

四、字符串

name = "zhangsan,my name is {name},and I'm {year} years old"

#将字符串首字母大写
print(name.capitalize())

#统计指定字符出现次数
print(name.count("a"))

#name居中，一共打印50个字符，不足用-补充
print(name.center(50,"-"))

#判断是否以...结尾
print(name.endswith("an"))

#find:查找字符串中第一次出现指定字符的位置
#[:]字符串切片
print(name[name.find("a"):7])

#format:格式化，映射方式传递
#format_map:后面使用字典传递

print(name.format_map({'name':'zhangsan','year':'20'}))

#判断是否为整数
print('354'.isdigit())

#使用指定连字符将列表中元素连接 连字符.join(list)
print('+'.join(['1','2','3']))

#从左边开始70个字符，不足用*补充
print( name.ljust(70,'*') )
#从右侧开始70个字符，不足用-补充
print(name.rjust(70,'-'))

#strip()去除两边的空格和回车，lstrip()去除左面的空格和回车
print('\nZhangsan'.lstrip())

#将预设的对应转换关系在新字符串中相对应，形成字典 
p=str.maketrans("abcdef",'123456')
#使用translate将字典应用于新的字符串
print("zhangsan".translate(p))

#replace('oldstr','newstr',times)
#将oldstr替换为newstr，从左到右替换times次
print('zhangsan'.replace('z','Z',1))

#将指定字符作为分割符分割字符串形成列表
print('zhangsan'.split('a'))

exit_mark= True
while exit_mark:
    #只打印出city_map中的key
    for i in city_map:
        print(i)
    choice = input("选择进入>>>:")
    if choice in city_map:
        while exit_mark:
            for i2 in city_map[choice]:
                print("\t",i2)

                    if choice3 in city_map[choice][choice2]:
                        for i4 in city_map[choice][choice2][choice3]:

                        choice4 = input("最后一层，按b返回上一级")
                        if choice4 == "b":
                            pass
                        elif choice4 =="q":
                            exit_mark = False
                    if choice3 == "b":
                        break
                    elif choice3 =="q":
                        exit_mark = False
            if choice2 == "b":
                break
            elif choice2 == "q":
                exit_mark = False

七、集合

集合重要的两个作用：

  1. 去重
  2. 关系测试
    集合是无序的

1. 集合的创建

#集合的创建
#通过set(list)创建
list_1 = [1,4,5,7,3,6,7,9]
list_1 = set(list_1)
#通过set([])创建
list_2 =set([2,6,0,66,22,8,4])
print(list_1,list_2)

2. 集合运算

#取交集
interlist_1=list_1.intersection(list_2)
print(interlist_1)

#取并集
unionlist_1 = list_1.union(list_2)
print(unionlist_1)

#差集
diflist_1 = list_1.difference(list_2)
print(diflist_1)

#子集、父集
#A.issubset(B):判断A是否为B的子集，返回True or False
#A.issuperset(B):判断A是否为B的父集，返回True or False
list_3 = set([1,3,7])
print(list_3.issubset(list_1))
print(list_1.issuperset(list_3))

#对称差集
print(list_1.symmetric_difference(list_2))

"""
符号表示
交集 &
并集 |
差集 -
对称差集 ^
"""

5. 例子

#high loop 高效
#实现创建将第十行的句子替换为----continue-----
f = open("yesterday",'r',encoding='utf-8')
count = 0
for line in f:
    if count ==9:
        print('--------continue--------')
        count+=1
        continue
    count += 1
    #读文件的时候换行符\n也被读进去，在逐行打印中去掉
    print(line.strip())
f.close()

#low loop 低效
#实现第十行句子替换为-------end------
#f.readlines:以字符串形式读入
#当文件很大时特别占用内存
f = open("yesterday",'r',encoding='utf-8')
for index,line in enumerate(f.readlines()):
    if index == 9:
        print('-------end------')
        continue
    print(line.strip())

九、字符编码

当编译语言使用系统默认编码(uft-8)时：

# Author:ZT
'''
py程序代码的默认编码是utf-8
py字符串是以unicode编码的
encode()默认采用系统编码将Unicode编码成utf-8
encode("gbk")编码成gbk
'''
import sys
#打印出系统的默认编码
print(sys.getdefaultencoding())
s="哈哈"
#将Unicode编码成gbk
s_gbk = s.encode("gbk")
print(s_gbk)
print(s.encode())
#将gbk转换成utf-8
#先解码成unicode,再编码成utf-8
gbk_utf8 = s_gbk.decode("gbk").encode("utf-8")
print(gbk_utf8)

当编译语言使用gbk编码时：

#将编译语言设置为gbk编码
#-*-coding:gbk-*-
# Author:ZT
"""
python3中进行encode时自动转换成byte类型
"""
import sys
#获取系统默认编码
print(sys.getdefaultencoding())
#保存到字符串的时候依旧是unicode编码
s="你好"
#编码成gbk
print(s.encode("gbk"))
print(s.encode("utf-8").decode("utf-8").encode("gb2312"))
#编码成utf-8->解码->编码成gb2312->解码，最终为unicode
print(s.encode("utf-8").decode("utf-8").encode("gb2312").decode("gb2312"))
