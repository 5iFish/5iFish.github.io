  1. 增

    #追加到最后 list.append
    names.append(“LeiHaidong”)
    #插入到所选位置之前 list.insert(numb,str)
    names.insert(1,”ChenRonghua”)
    names.insert(3,”XinZhiyu”)

  2. 删

    #delete1 list.remove(str)
    names.remove(“ChenRonghua”)
    #delete2 del list[numb]
    del names[1]
    #delete3 list.pop()
    names.pop() #删除最后一个

  3. 改

    #修改 list[numb]=str
    names[2]=”XieDi”
    print(names)

  4. 查

    #查找 list.index(str)返回位置
    print(names.index(“XieDi”))

  5. 切片

    names=[‘张三’,’Guyun’,’Xiangpeng’,’Alex’]
    print(names[1:2])
    ‘’’
    [a:b]取值为[a,b)左闭右开区间
    可以将:理解为省略的都取
    显示结果
    [‘Guyun’]
    ‘’’
    #取最后一个值
    print(names[-1])
    #取最后两个值
    print(names[-1:-3])#错误取法
    ‘’’
    结果为
    []
    应该是自左向右取值
    ‘’’
    print(names[-3:-1])#错误取法
    ‘’’
    结果为
    [‘Guyun’, ‘XiangPeng’]
    取不到最后一位
    ‘’’
    print(names[-2:])#正确取法
    #自前向后取3个
    print(names[:3])
    names2=[‘张三’,’Alex’,’Guna’]
    #步长切片list[起:终:步长]
    print(names[0:-1:2])
    print(names[::2])

  6. copy

    import copy

    names = [“ZhangYang”,”Guoyun “,[“jack”,”Anny”]]
    names2 = names.copy()
    print(names)
    print(names2)

    #copy只复制一级列表，浅copy
    #names中的二级列表[“jack”,”Anny”]只复制了二级列表的id
    #当通过names[2][1]修改时，names2中的二级列表也会被修改
    names[1]=”郭云”
    names[2][1]=”安妮”
    print(names)
    print(names2)
    #copy.deepcopy:深copy
    #涉及到二级列表时创建新的列表id并创建二级列表
    #改变names的二级列表时，不改变names2的内容
    names3 = copy.deepcopy(names)
    names[2][1]=”anny”
    print(names)
    print(names3)
